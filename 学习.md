index.jx:


import {AppRegistry} from 'react-native';  //导入AppRegistry组件进行首页的注册
import App from './App';
import {name as appName} from './app.json';   

AppRegistry.registerComponent(appName, () => App);  //进行首页的注册 ，appName这是项目的名称，这个不能改，像app这样的事第二个参数，是指将那个
页面指定为首页
